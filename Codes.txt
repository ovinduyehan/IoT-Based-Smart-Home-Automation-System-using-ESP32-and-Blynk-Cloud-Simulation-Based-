#define BLYNK_PRINT Serial

#define BLYNK_TEMPLATE_ID "YOUR_TEMPLATE_ID"
#define BLYNK_TEMPLATE_NAME "Smart Home Automation"
#define BLYNK_AUTH_TOKEN "KPYVIRarvi58Ic1zZFt81etZoCceS9xw"

#include <WiFi.h>
#include <BlynkSimpleEsp32.h>

char ssid[] = "Wokwi-GUEST";
char pass[] = "";

#define LIGHT_PIN 2
#define FAN_PIN 4
#define APPLIANCE_PIN 5

BLYNK_WRITE(V0) { digitalWrite(LIGHT_PIN, param.asInt()); }
BLYNK_WRITE(V1) { digitalWrite(FAN_PIN, param.asInt()); }
BLYNK_WRITE(V2) { digitalWrite(APPLIANCE_PIN, param.asInt()); }

void setup() {
  Serial.begin(115200);

  pinMode(LIGHT_PIN, OUTPUT);
  pinMode(FAN_PIN, OUTPUT);
  pinMode(APPLIANCE_PIN, OUTPUT);

  WiFi.begin(ssid, pass);

  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }

  Serial.println("\nWiFi Connected");

  Serial.println("Connecting to Blynk...");

  Blynk.config(BLYNK_AUTH_TOKEN);
  Blynk.connect();
}

void loop() {
  Blynk.run();
}
